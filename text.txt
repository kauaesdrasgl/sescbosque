O que é Docker?
Docker é uma ferramenta que permite rodar programas (como bancos de dados, Ollama, etc.) dentro de "caixas" isoladas chamadas containers.
É muito usado porque é mais estável, fácil de configurar e evita vários problemas de instalação direta no Windows.

1. Verifique os requisitos do seu PC

Windows 10 ou 11 (64 bits) Pro, Enterprise ou Education (Home geralmente tem limitações)

Execute o arquivo Docker Desktop Installer.exe que foi baixado
Durante a instalação:
Marque a opção "Use WSL 2 instead of Hyper-V" (recomendado)
Clique em OK e espere instalar

Após instalar, o Docker Desktop vai abrir automaticamente.

// Função auxiliar para buscar dados
async function fetchFromSupabase(table) {
  if (!supabaseClient) {
    console.warn("Supabase não inicializado")
    return null
  }
  try {
    const { data, error } = await supabaseClient
      .from(table)
      .select('*')
      .order('created_at', { ascending: false })

    if (error) throw error
    console.log(`✅ Dados carregados de ${table}:`, data?.length || 0, "itens")
    return data
  } catch (error) {
    console.error(`Erro ao buscar ${table}:`, error)
    return null
  }
}

export { supabaseClient, initSupabase, fetchFromSupabase }

HTML<script>
async function uploadImagem(file) {
  if (!file) return null;

  try {
    const fileName = Date.now() + '-' + file.name.replace(/\s+/g, '-');

    const { data, error } = await supabaseClient.storage
      .from('produtos')
      .upload(fileName, file, {
        cacheControl: '3600',
        upsert: true
      });

    if (error) throw error;

    const { data: urlData } = supabaseClient.storage
      .from('produtos')
      .getPublicUrl(fileName);

    console.log("✅ Imagem enviada:", urlData.publicUrl);
    return urlData.publicUrl;

  } catch (error) {
    console.error("Erro no upload:", error);
    toast("Erro ao fazer upload da imagem", "red");
    return null;
  }
}

// Atualize a função salvarItem() para incluir upload
